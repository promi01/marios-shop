    ```typescript
    import { formatPrice, formatItemCount } from '@/lib/format';

    export interface ResolvedItem {
      brand: string;
      name: string;
      typeLabel: string; // Greek (Σφραγισμένο / Ανοιγμένο / Decant) per CONTEXT D-25
      size_ml: number;
      price: number;
      quantity: number;
    }

    /**
     * Format the cart as Messenger-ready plain text per UI-SPEC §Copy-to-Messenger Format.
     * Returns "" for an empty list (defensive — the Copy button is disabled when cart is empty).
     */
    export function formatOrderText(items: ResolvedItem[]): string {
      if (items.length === 0) return '';

      const header = 'Παραγγελία — Marios Shop';

      const itemBlocks = items.map((item, index) => {
        const subtotal = item.price * item.quantity;
        const firstLine = `${index + 1}. ${item.brand} — ${item.name}`;
        const secondLine = `   ${item.typeLabel} ${item.size_ml}ml — ${formatPrice(item.price)} × ${item.quantity} = ${formatPrice(subtotal)}`;
        return `${firstLine}\n${secondLine}`;
      });

      const totalQty = items.reduce((sum, i) => sum + i.quantity, 0);
      const totalPrice = items.reduce((sum, i) => sum + i.price * i.quantity, 0);
      const footer = `Σύνολο: ${formatPrice(totalPrice)} — ${formatItemCount(totalQty)}`;

      return [header, '', ...itemBlocks.flatMap((b) => [b, '']), footer].join('\n');
    }
    ```
    Format rules baked in:
    - Header line `Παραγγελία — Marios Shop` followed by a blank line.
    - Each item: `{n}. {brand} — {name}` on first line, then 3-space indent + `{TypeLabelGr} {size_ml}ml — {price}€ × {qty} = {subtotal}€` on the next line.
    - Blank line between items (achieved via the `flatMap` interleaving an empty string after each block; the final empty string before the footer becomes the separator).
    - Footer: `Σύνολο: {total}€ — {count} τεμάχια` using `formatItemCount` to handle singular vs plural.
    - Per CONTEXT D-25: `fill_pct` is intentionally NOT part of `ResolvedItem`'s interface and never appears in the output.
    - Per CONTEXT D-26: prices via `formatPrice` (suffix euro, no space).

    **1. `lib/clipboard.ts`** — Primary path + textarea fallback per COPY-06 + CONTEXT D-24:
    ```typescript
    'use client';

    /**
     * Copy `text` to the OS clipboard.
     *
     * Primary path: navigator.clipboard.writeText (modern browsers, requires HTTPS or localhost).
     * Fallback: hidden textarea + document.execCommand('copy') for older browsers
     * or non-secure contexts that lack the Clipboard API.
     *
     * Returns true on success, false on failure.
     */
    export async function copyToClipboard(text: string): Promise<boolean> {
      // Primary path
      if (typeof navigator !== 'undefined' && navigator.clipboard?.writeText) {
        try {
          await navigator.clipboard.writeText(text);
          return true;
        } catch {
          // Fall through to legacy path below.
        }
      }

      // Legacy fallback
      if (typeof document === 'undefined') return false;
      const textarea = document.createElement('textarea');
      textarea.value = text;
      textarea.setAttribute('readonly', '');
      textarea.style.position = 'fixed';
      textarea.style.top = '0';
      textarea.style.left = '0';
      textarea.style.opacity = '0';
      textarea.style.pointerEvents = 'none';
      document.body.appendChild(textarea);
      textarea.select();
      let success = false;
      try {
        success = document.execCommand('copy');
      } catch {
        success = false;
      }
      document.body.removeChild(textarea);
      return success;
    }
    ```
    Decisions:
    - Marked `'use client'` because both branches touch `navigator` / `document`. In practice it's only called from event handlers, but the directive future-proofs the module.
    - Primary path is `navigator.clipboard.writeText` per COPY-06; failure (rejected promise) falls through to the legacy path — same module, no separate error handling needed at call sites.
    - Legacy fallback uses the off-screen textarea trick. `position: fixed; top: 0; left: 0; opacity: 0` keeps it invisible without breaking selection. `readonly` prevents iOS keyboard pop. Cleanup is in a try/finally pattern.
    - Per CONTEXT D-24: NO toast wording difference between the two paths — both paths return `true`/`false` and the caller renders the SAME success/error toast.

    **2. `components/copy-to-messenger-button.tsx`** — The button per COPY-01, COPY-02, COPY-05, COPY-06, COPY-08:
    ```tsx
    'use client';

    import { toast } from 'sonner';
    import { Button } from '@/components/ui/button';
    import { useCartStore } from '@/lib/cart-store';
    import { getProductById, getVariant } from '@/lib/inventory';
    import { formatTypeLabel } from '@/lib/format';
    import { formatOrderText, type ResolvedItem } from '@/lib/copy-format';
    import { copyToClipboard } from '@/lib/clipboard';

    export function CopyToMessengerButton() {
      const items = useCartStore((s) => s.items);
      const isHydrated = useCartStore((s) => s.isHydrated);

      // Disable per COPY-08 when the cart is empty (or before hydration).
      const isEmpty = !isHydrated || items.length === 0;

      const handleCopy = async () => {
        const resolved: ResolvedItem[] = items
          .map((item) => {
            const product = getProductById(item.product_id);
            const variant = product ? getVariant(product, item.variant_id) : undefined;
            if (!product || !variant) return null;
            return {
              brand: product.brand,
              name: product.name,
              typeLabel: formatTypeLabel(variant.type),
              size_ml: variant.size_ml,
              price: variant.price,
              quantity: item.quantity,
            };
          })
          .filter((x): x is ResolvedItem => x !== null);

        if (resolved.length === 0) return;

        const text = formatOrderText(resolved);
        const ok = await copyToClipboard(text);

        if (ok) {
          toast.success('Αντιγράφηκε!');
        } else {
          toast.error('Δεν αντιγράφηκε — δοκιμάστε ξανά');
        }
      };

      return (
        <Button
          type="button"
          onClick={handleCopy}
          disabled={isEmpty}
          aria-disabled={isEmpty}
          className="w-full"
          data-slot="copy-cta"
        >
          📋 Αντιγραφή για Messenger
        </Button>
      );
    }
    ```
    Decisions:
    - Subscribes to `items` + `isHydrated`. While `!isHydrated` the button is disabled (avoids a flash of enabled-then-disabled and prevents trying to copy a not-yet-rehydrated cart).
    - `handleCopy` resolves the persisted items into `ResolvedItem[]` per CONTEXT D-06: brand/name/type/size/price are read fresh from `inventory.ts`. The `typeLabel` field is the Greek label via `formatTypeLabel` (per CONTEXT D-25).
    - If, after resolution, the list is empty (every item points to a deleted product — extreme edge case), the function returns silently without firing a toast. The button being `disabled={isEmpty}` makes this branch unreachable in normal usage; it's defensive.
    - On success: `toast.success('Αντιγράφηκε!')` per UI-SPEC §Copywriting Contract (Toast — copy success).
    - On failure: `toast.error('Δεν αντιγράφηκε — δοκιμάστε ξανά')` per UI-SPEC §Copywriting Contract (Toast — copy failure).
    - Per COPY-08: `disabled={isEmpty}`. shadcn `<Button disabled>` automatically applies the disabled-state styling from UI-SPEC §6 (`bg-neutral-200 text-neutral-400 cursor-not-allowed`); if the shadcn build does not produce that exact visual, add `className="w-full disabled:bg-neutral-200 disabled:text-neutral-400 disabled:cursor-not-allowed"` explicitly.
    - Label `📋 Αντιγραφή για Messenger` — the ONE emoji allowed per UI-04 / COPY-01.
